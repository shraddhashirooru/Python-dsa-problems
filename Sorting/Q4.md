```
class Solution:
    def sortArr(self, arr):
        def mergesort(left,right):
            m=len(left)
            n=len(right)
            i=0
            j=0
            s=[]
            while i<m and j<n:
                if left[i]<right[j]:
                    s.append(left[i])
                    i+=1
                else:
                    s.append(left[j])
                    j=j+1
            while i<m:
                s.append(left[i])
                i+=1
            while j<n:
                s.append(right[j])
                j+=1
            return s  
        
        if len(arr)<=1:
            return arr
        mid=len(arr)//2
        left=self.sortArr(arr[:mid])
        right=self.sortArr(arr[mid:])
        return mergesort(left,right)
```
